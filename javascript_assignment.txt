// Part 1: Basics
function greetUser() {
  let name = document.getElementById("userInput").value;
  if (name === "") {
    document.getElementById("greeting").innerText = "Please enter your name!";
  } else {
    document.getElementById("greeting").innerText = "Hello, " + name;
  }
}

// Part 2: Functions
function calculateTotal(price, qty) {
  return price * qty;
}
function showTotal() {
  let total = calculateTotal(50, 2);
  document.getElementById("totalDisplay").innerText = "Total: $" + total;
}

// Part 3: Loops
function countdown() {
  let text = "";
  for (let i = 5; i > 0; i--) {
    text += i + " ";
  }
  document.getElementById("countdownResult").innerText = text;
}

// Part 4: DOM
document.getElementById("toggleBtn").addEventListener("click", function() {
  document.body.classList.toggle("dark");
});
function addItem() {
  let li = document.createElement("li");
  li.innerText = "New Item";
  document.getElementById("itemList").appendChild(li);
}
